# WPU Faculty System - Environment Configuration
# ================================================
# 
# This file documents all environment variables used by the system.
# Configure these on your server for production deployment.
# 
# HOW TO USE:
# -----------
# 
# Option 1: Apache - Add to your VirtualHost configuration:
#   SetEnv DB_HOST "localhost"
#   SetEnv DB_NAME "wpu_faculty_system"
#   SetEnv DB_USER "your_db_user"
#   SetEnv DB_PASS "your_db_password"
#   SetEnv PRODUCTION "true"
#   (etc for other variables)
#
# Option 2: PHP-FPM pool configuration (/etc/php/X.X/fpm/pool.d/www.conf):
#   env[DB_HOST] = localhost
#   env[DB_NAME] = wpu_faculty_system
#   env[DB_USER] = your_db_user
#   env[PRODUCTION] = true
#   (etc)
#
# Option 3: System environment (Linux/bash - add to /etc/environment or ~/.bashrc):
#   export DB_HOST="localhost"
#   export DB_NAME="wpu_faculty_system"
#   export PRODUCTION="true"
#   (etc)
#
# Option 4: Windows - System Environment Variables:
#   1. Open System Properties → Advanced → Environment Variables
#   2. Add each variable under System Variables
#
# Option 5: IIS - Application Pool Environment Variables:
#   Configure via IIS Manager → Application Pools → Advanced Settings
#
